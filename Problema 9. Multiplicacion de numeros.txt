Realizar un programa que muestre en pantalla la tabla de multiplicar de un número a preguntar.

.model small
.stack 100h
.data

	mens1 db "Dame un valor"
	valor db 0
	conta db 1
.code

	mov dx,offset mens1
	mov ah,09h
	int 21h
	mov ah,8h
	int 21
	mov valor,al :Movemos el numero que nos dieron a al

multiplicacion:
	mov valor,al
  
	mul conta 
	mov ah,02h :Salto de linea en salida de pantalla
	mov dl,0xA 
	int 21h 

	mov ah,02h :Salto de linea en salida de pantalla
	mov dl,0xD 
	int 21h 
 	
	mov ah,02h : Imprime el numero ingresado
	mov dl,valor 
	int 21h 

	mov ah,02h : Imprime la letra X
	mov dl,'X' 
	int 21h 
	
	mov ah,02h : Imprime el simbolo = 
	mov dl,'=' 
	int 21h 

	inc conta

	cmp conta,10 
	jbe multiplicacion :Si el numero es menor a 10 lleva a multiplicacion
fin:end
