1.- Programa donde preguntamos la calificacion, donde si esta reprobado añadiremos 20pts

.modelo small

.stack 100h

.data
	
	mens1 db "¿Cuál es tu calificación?"
	
	mens2 db "Estas aprobado"
	
	mens3 db "Estas reprobado, añadiendo 20pts"
	
	calif db 0

.code
	
	mov dx, offset mens1
	
	mov ah, 09h 			
	
	int 21h  	
			
	

	mov ah, 8h
	int 21h 	
		
	
	mov calif, al
	
	cmp calif, 70			
	
		jge aprobado

		add calif,20		
		cmp calif, 70
			jge aprobado
				mov dx, offset mens3
				mov ah, 09h
				int 21h
				jmp fin
	aprobado:
		mov dx, offset mens2
		mov ah, 09h
		int 21h

fin: end.
